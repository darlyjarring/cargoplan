async function cargarDaesTPG() { const sem = document.getElementById('txtSemana').value; const emp = document.getElementById('selEmpresa').value; const tbody = document.getElementById('listaDaes'); if(!sem) return alert("Ingrese semana"); tbody.innerHTML = 'Buscando DAES en terminales TPG...'; // 1. Consultamos TODAS las DAES de esa semana y empresa const { data, error } = await _supabase .from('reporte_daes') .select('*') .eq('semana', sem) .eq('empresa', emp); if (error) { tbody.innerHTML = `Error: ${error.message}`; return; } // 2. Filtro Ultra-Flexible para detectar TPG / INARPI // Esto detectará: "TPG", "INARPI", "INARPI / TPG", "INARPI/TPG", etc. const filtrados = data.filter(d => { const termOriginal = (d.puerto || "").toUpperCase(); // Quitamos espacios para comparar: "INARPI / TPG" -> "INARPI/TPG" const termLimpia = termOriginal.replace(/\s+/g, ''); return termLimpia.includes("TPG") || termLimpia.includes("INARPI") || termLimpia.includes("TPG-4"); }); tbody.innerHTML = ''; if(filtrados.length === 0) { tbody.innerHTML = 'No se encontraron DAES vinculadas a TPG/INARPI para esta semana.'; return; } // 3. Renderizar resultados filtrados.forEach(d => { const tr = document.createElement('tr'); tr.innerHTML = ` ${d.booking} ${d.dae} ${d.puerto} ${d.puerto_destino || d.destino || ''} `; tbody.appendChild(tr); }); }
